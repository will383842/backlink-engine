model CampaignTag {
  campaignId Int
  tagId      Int
  createdAt  DateTime @default(now())

  campaign Campaign @relation(fields: [campaignId], references: [id], onDelete: Cascade)
  tag      Tag      @relation(fields: [tagId], references: [id], onDelete: Cascade)

  @@id([campaignId, tagId])
  @@index([campaignId])
  @@index([tagId])

  @@map("campaign_tags")
}
